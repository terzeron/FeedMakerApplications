function setEventCookie( name, value, expiredays ) {
    var todayDate = new Date();

    todayDate.setDate(todayDate.getDate());
    var naxHours = 23 - todayDate.getHours();
    var naxMinutes = 60 - todayDate.getMinutes();
    var naxSeconds = 60 - todayDate.getSeconds();

    todayDate.setHours(todayDate.getHours() + naxHours);
    todayDate.setMinutes(todayDate.getMinutes() + naxMinutes-1);
    todayDate.setSeconds(todayDate.getSeconds() + naxSeconds);
    document.cookie = name + "=" + escape( value ) + "; path=/; expires=" + todayDate.toGMTString() + ";"
}
