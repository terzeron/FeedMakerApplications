		if ($(v).hasClass('on')) {
			$(v).removeClass('on');
			$('#web_left .view_'+$(v).attr('id')).hide();
		}
		if ($(v).attr('id')==data) {
			$(v).addClass('on');
			$('#web_left .view_'+data).show();
		}
	});
}
$(function() {
	viewTop10(view_status.toon.top10 || 'ma1');

	$('#nav_fin li').on('click', function(e) {
		if($(this).hasClass('on')==false) {
			var data = $(this).attr('id');
			viewTop10(data);
			view_status.toon.top10 = data;
			Cookies.set('view_status', view_status);
		}
	});
});
</script>
