2. 다수의 슬라이드 입력 vba 코드

Sub InsertModuleContentsToTitleOnlySlides()
    Dim ppt As Presentation
    Dim sld As Slide
    Dim i As Integer
    Dim moduleText As String
    Dim titleLayout As CustomLayout
    Dim slideIndex As Integer

    Set ppt = ActivePresentation

    ' 제목만 있는 레이아웃 찾기 (기본적으로 "제목만"이라는 이름을 가정)
    For i = 1 To ppt.Designs(1).SlideMaster.CustomLayouts.Count
        If ppt.Designs(1).SlideMaster.CustomLayouts(i).Name = "제목만" Then
            Set titleLayout = ppt.Designs(1).SlideMaster.CustomLayouts(i)
            Exit For
        End If
    Next i

    If titleLayout Is Nothing Then
        MsgBox "제목만 레이아웃을 찾을 수 없습니다."
        Exit Sub
    End If

    ' 18개의 모듈 텍스트를 삽입
    For i = 1 To 18
        moduleText = "{모듈" & i & " 내용}"

        slideIndex = ppt.Slides.Count + 1
        Set sld = ppt.Slides.AddSlide(slideIndex, titleLayout)
        sld.Shapes.Title.TextFrame.TextRange.Text = moduleText
    Next i
End Sub

3. 변수 치환 후 슬라이드 내용 삽입 프롬프트

🧠 목적:
PowerPoint 프레젠테이션의 슬라이드 및 테이블에 포함된 텍스트 중 특정 플레이스홀더({모듈1 내용} ~ {모듈13 내용})를 실제 교육 내용 또는 마케팅 기법 내용으로 자동 대체하는 VBA 코드를 작성하는 것.

📌 프롬프트 (Prompt):

너는 VBA 전문가야. 다음 요구사항에 맞는 PowerPoint 매크로 코드를 작성해 줘.

📍 목적:
PowerPoint 슬라이드에서 텍스트 또는 표 안에 포함된 플레이스홀더 변수들을 실제 텍스트로 자동 치환하는 매크로를 만들고 싶어. 플레이스홀더는 {모듈1 내용}부터 {모듈18 내용}까지이며, 각 변수는 3줄짜리 멀티라인 텍스트로 대체될 거야.

📍 전제 조건:

현재 PowerPoint 프레젠테이션 파일(ActivePresentation)의 모든 슬라이드를 대상으로 함

각 슬라이드에 있는 모든 Shape를 반복하면서 텍스트를 가진 경우 해당 텍스트를 치환함

Shape가 표(Table)일 경우에는 표 안의 모든 Cell을 순회하여 텍스트 치환을 수행함

교체할 텍스트는 replacements라는 Dictionary 객체를 사용해 구성함

📍 요구 사항:

다음과 같은 구조로 replacements 딕셔너리를 정의해 줘:

key: {모듈1 내용} ~ {모듈18 내용}

value: 각 key에 대응되는 텍스트는 vbCrLf를 포함한 **3줄짜리** 텍스트

{모듈1 내용}~{모듈9 내용}은 **문서 작성 교육 프로그램 내용으로 랜덤으로 발췌된 주제**를 3줄 설명으로 구성해줘,

{모듈10 내용}~{모듈18 내용}은 **마케팅 기법 주제에서 랜덤으로 발췌된 주제**를 3줄 설명으로 구성해줘 (예: SNS 마케팅, SEO 전략 등)

텍스트 치환은 다음을 포함해야 해:

일반 텍스트 상자에서 플레이스홀더 텍스트가 존재하면 해당 키워드를 찾아 value로 치환

표(Table) 안의 셀(Cell) 안에 텍스트가 존재하면 동일하게 치환

치환 완료 후에는 MsgBox로 “모든 변수 값이 성공적으로 대체되었습니다.” 메시지를 출력해 줘.

가독성과 유지보수를 위해 들여쓰기와 주석도 명확히 작성해줘.

📍 결과물 형태:

PowerPoint에서 실행 가능한 Sub 프로시저 형태의 VBA 코드

이름은 Sub ReplaceVariables()

예시로 {모듈6 내용}은 "SNS 마케팅 전략의 이해" 등으로 시작하고 vbCrLf로 줄을 구분한 멀티라인 텍스트여야 해.

이 요구에 맞게 전체 코드를 작성해줘.

4. 파워포인트 vba 코드

Option Explicit

'=========================================================
' "레이아웃: 제목만" 슬라이드를 18장 생성(또는 확보)한 뒤,
' 각 슬라이드의 제목(Title) 자리에 아래 텍스트를 순차 입력
'  {모듈1 내용} ~ {모듈18 내용}
'
' ※ 중괄호 포함, 그대로 삽입됩니다.
'=========================================================
Public Sub Insert_ModuleContents_ToTitleOnly()
    On Error GoTo EH

    Dim i As Long
    Dim sld As Slide
    Dim needCount As Long
    Dim titles(1 To 18) As String

    '---------------------------------------------
    ' 1) 입력할 텍스트(중괄호 포함, 그대로)
    '---------------------------------------------
    titles(1) = "{모듈1 내용}"
    titles(2) = "{모듈2 내용}"
    titles(3) = "{모듈3 내용}"
    titles(4) = "{모듈4 내용}"
    titles(5) = "{모듈5 내용}"
    titles(6) = "{모듈6 내용}"
    titles(7) = "{모듈7 내용}"
    titles(8) = "{모듈8 내용}"
    titles(9) = "{모듈9 내용}"
    titles(10) = "{모듈10 내용}"
    titles(11) = "{모듈11 내용}"
    titles(12) = "{모듈12 내용}"
    titles(13) = "{모듈13 내용}"
    titles(14) = "{모듈14 내용}"
    titles(15) = "{모듈15 내용}"
    titles(16) = "{모듈16 내용}"
    titles(17) = "{모듈17 내용}"
    titles(18) = "{모듈18 내용}"

    '---------------------------------------------
    ' 2) "제목만(ppLayoutTitleOnly)" 레이아웃으로 18장 생성
    '    - 기존 슬라이드가 18장 미만이면 추가 생성
    '    - 기존 슬라이드가 18장 이상이면 앞 18장에 적용
    '---------------------------------------------
    needCount = 18

    If ActivePresentation Is Nothing Then
        MsgBox "열려 있는 프레젠테이션이 없습니다.", vbExclamation
        Exit Sub
    End If

    ' 부족하면 슬라이드 추가 (제목만 레이아웃)
    Do While ActivePresentation.Slides.Count < needCount
        ActivePresentation.Slides.Add ActivePresentation.Slides.Count + 1, ppLayoutTitleOnly
    Loop

    '---------------------------------------------
    ' 3) 1~18번 슬라이드의 제목(Title) 에 순차 입력
    '---------------------------------------------
    For i = 1 To needCount
        Set sld = ActivePresentation.Slides(i)

        ' 해당 슬라이드를 "제목만" 레이아웃으로 강제 적용
        sld.Layout = ppLayoutTitleOnly

        ' 제목 플레이스홀더가 없을 수 있으므로 예외 처리
        If Not sld.Shapes.Title Is Nothing Then
            sld.Shapes.Title.TextFrame.TextRange.Text = titles(i)
        Else
            ' Title이 없다면, 텍스트 상자를 새로 만들어 상단에 배치
            Dim tb As Shape
            Set tb = sld.Shapes.AddTextbox( _
                        Orientation:=msoTextOrientationHorizontal, _
                        Left:=50, Top:=30, Width:=620, Height:=80)
            tb.TextFrame.TextRange.Text = titles(i)
        End If
    Next i

    MsgBox "입력이 완료되었습니다. (슬라이드 1~18 제목에 {모듈n 내용} 삽입)", vbInformation
    Exit Sub

EH:
    MsgBox "오류가 발생했습니다." & vbCrLf & _
           "오류번호: " & Err.Number & vbCrLf & _
           "오류내용: " & Err.Description, vbExclamation
End Sub
