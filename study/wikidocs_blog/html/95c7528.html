1.1 OpenAI API 연동

📖 스토리

"먼저 LLM을 연동해보자," 레이나가 말했다.

from openai import OpenAI
from typing import List, Dict
import os

class LLMService:
    """
    LLM API 통합 서비스

    OpenAI GPT-4o 사용
    """

    def __init__(self, api_key: str = None):
        """
        Args:
            api_key: OpenAI API 키 (없으면 환경변수에서 가져옴)
        """
        self.api_key = api_key or os.getenv('OPENAI_API_KEY')
        self.client = OpenAI(api_key=self.api_key)
        self.model = "gpt-4o"  # 빠르고 저렴

        print(f"🤖 LLM 서비스 초기화")
        print(f"   모델: {self.model}")

    def generate(
        self,
        prompt: str,
        max_tokens: int = 500,
        temperature: float = 0.3
    ) -> Dict:
        """
        답변 생성

        Args:
            prompt: 프롬프트
            max_tokens: 최대 토큰 수
            temperature: 창의성 (0~1, 낮을수록 일관적)

        Returns:
            {
                'answer': 생성된 답변,
                'usage': 토큰 사용량,
                'cost': 비용 (달러)
            }
        """

        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "user", "content": prompt}
            ],
            max_tokens=max_tokens,
            temperature=temperature
        )

        # 비용 계산 (GPT-4o 기준: 입력 $2.5/1M, 출력 $10/1M)
        input_tokens = response.usage.prompt_tokens
        output_tokens = response.usage.completion_tokens

        input_cost = (input_tokens / 1_000_000) * 2.5
        output_cost = (output_tokens / 1_000_000) * 10.0
        total_cost = input_cost + output_cost

        return {
            'answer': response.choices[0].message.content,
            'usage': {
                'input_tokens': input_tokens,
                'output_tokens': output_tokens,
                'total_tokens': response.usage.total_tokens
            },
            'cost': total_cost
        }

1.2 RAG Prompt 설계

📖 스토리

"이제 중요한 부분이야," 레이나가 말했다. "프롬프트 디자인."

효과적인 RAG Prompt 구조

1️⃣ 역할 정의
   "당신은 DataPulse 사내 문서 전문가입니다."

2️⃣ 지시사항
   "제공된 문서만을 기반으로 답변하세요."
   "문서에 없는 내용은 추측하지 마세요."

3️⃣ 컨텍스트 제공
   "다음 문서들을 참고하세요:
    [검색된 문서들]"

4️⃣ 질문
   "질문: [사용자 질문]"

5️⃣ 답변 형식
   "답변은 다음 형식으로 작성하세요:
    - 명확하고 간결하게
    - 출처를 명시
    - 확실하지 않으면 '문서에 해당 정보 없음' 명시"

준혁이 구현을 시작했다.

class RAGPromptBuilder:
    """
    RAG 프롬프트 생성기

    효과적인 RAG 프롬프트 자동 생성
    """

    def __init__(self):
        # 시스템 프롬프트 (역할 정의)
        self.system_prompt = """당신은 DataPulse 사내 문서 전문가입니다.

**중요 지시사항:**
1. 제공된 문서만을 기반으로 답변하세요
2. 문서에 없는 내용은 절대 추측하지 마세요
3. 확실하지 않으면 "문서에 해당 정보가 없습니다"라고 명시하세요
4. 모든 답변에 출처를 명시하세요 (문서명, 페이지)
5. 답변은 명확하고 간결하게 작성하세요
6. 단계별 설명이 필요한 경우 번호를 매기세요"""

    def build_prompt(
        self,
        query: str,
        documents: List[Dict],
        max_context_length: int = 3000
    ) -> str:
        """
        RAG 프롬프트 생성

        Args:
            query: 사용자 질문
            documents: 검색된 문서들 [{'document': str, 'source': str}, ...]
            max_context_length: 최대 컨텍스트 길이 (토큰 절약)

        Returns:
            완성된 프롬프트
        """

        # 컨텍스트 구성
        context_parts = []
        current_length = 0

        for i, doc in enumerate(documents, 1):
            doc_text = doc['document']
            source = doc.get('source', f'문서{i}')

            # 길이 체크
            if current_length + len(doc_text) > max_context_length:
                # 남은 공간만큼만 추가
                remaining = max_context_length - current_length
                if remaining > 100:  # 최소 100자는 있어야 의미 있음
                    doc_text = doc_text[:remaining] + "..."
                else:
                    break

            context_parts.append(f"""
[문서 {i}] 출처: {source}
{doc_text}
""")
            current_length += len(doc_text)

        context = "\n".join(context_parts)

        # 전체 프롬프트 조합
        prompt = f"""{self.system_prompt}

**참고 문서:**
{context}

**질문:**
{query}

**답변:**
위 문서들을 기반으로 질문에 답변해주세요. 답변 끝에는 반드시 출처를 명시하세요."""

        return prompt

    def build_prompt_with_examples(
        self,
        query: str,
        documents: List[Dict]
    ) -> str:
        """
        Few-shot 예시가 포함된 프롬프트

        더 나은 답변 품질을 위해 예시 추가
        """

        examples = """
**좋은 답변 예시:**

질문: "복지 신청은 어떻게 하나요?"
답변: "복지 신청은 다음과 같이 진행됩니다:
1. 인사팀 포털(hr.datapulse.com) 접속
2. '복지 신청' 메뉴 클릭
3. 원하는 복지 선택
4. 신청서 작성 후 제출

출처: 복지제도안내.pdf (5페이지)"

질문: "회사 창립일이 언제인가요?"
답변: "죄송합니다. 제공된 문서에 회사 창립일 정보가 없습니다."
"""

        # 기본 프롬프트에 예시 추가
        base_prompt = self.build_prompt(query, documents)

        # 예시를 시스템 프롬프트 뒤에 삽입
        parts = base_prompt.split("**참고 문서:**")
        prompt_with_examples = parts[0] + examples + "\n**참고 문서:**" + parts[1]

        return prompt_with_examples

2.1 Citation 자동 생성

📖 스토리

화요일 오후, 재석이 중요한 지적을 했다.

"답변은 좋은데," 재석이 말했다. "출처가 정확한지 어떻게 확인하죠? LLM이 거짓 출처를 만들 수도 있잖아요."

"좋은 지적이에요," 레이나가 답했다. "Citation Verification이 필요하죠."

준혁이 구현을 시작했다.

class CitationSystem:
    """
    출처 인용 시스템

    답변의 출처를 자동으로 검증하고 링크 생성
    """

    def __init__(self):
        # 문서 메타데이터 저장
        self.document_metadata = {}

    def register_documents(self, documents: List[Dict]):
        """
        문서 메타데이터 등록

        Args:
            documents: [
                {
                    'doc_id': 'doc_001',
                    'title': '복지제도안내.pdf',
                    'url': 'https://datapulse.com/docs/welfare.pdf',
                    'page': 3
                },
                ...
            ]
        """
        for doc in documents:
            doc_id = doc['doc_id']
            self.document_metadata[doc_id] = doc

    def extract_citations_from_answer(self, answer: str) -> List[str]:
        """
        답변에서 출처 추출

        예: "출처: 복지제도안내.pdf (5페이지)" → ['복지제도안내.pdf']
        """
        import re

        # 출처 패턴 찾기
        citation_pattern = r'출처[:\s]*([^\n]+)'
        matches = re.findall(citation_pattern, answer)

        citations = []
        for match in matches:
            # 문서명 추출 (파일명 또는 제목)
            doc_names = re.findall(r'([가-힣a-zA-Z0-9_]+\.(?:pdf|docx|txt|md))', match)
            citations.extend(doc_names)

        return list(set(citations))  # 중복 제거

    def add_citation_links(
        self,
        answer: str,
        source_documents: List[Dict]
    ) -> Dict:
        """
        답변에 클릭 가능한 출처 링크 추가

        Args:
            answer: LLM이 생성한 답변
            source_documents: 검색된 원본 문서들

        Returns:
            {
                'answer_with_links': 링크가 추가된 답변,
                'citations': [출처 정보 리스트]
            }
        """

        citations = []

        # 각 문서에 대해 링크 생성
        for i, doc in enumerate(source_documents, 1):
            source = doc.get('source', f'문서{i}')
            doc_id = doc.get('doc_id', f'doc_{i}')

            # 메타데이터에서 URL 찾기
            metadata = self.document_metadata.get(doc_id, {})
            url = metadata.get('url', '#')
            title = metadata.get('title', source)

            citations.append({
                'number': i,
                'title': title,
                'url': url,
                'source': source
            })

        # 답변에 각주 형식으로 번호 추가
        answer_with_links = answer

        # "출처:" 부분을 각주 링크로 교체
        if "출처:" in answer or "**출처:**" in answer:
            # 기존 출처 부분 제거
            answer_parts = answer.split("**출처:**")
            if len(answer_parts) > 1:
                main_answer = answer_parts[0].strip()
            else:
                answer_parts = answer.split("출처:")
                main_answer = answer_parts[0].strip()

            # 새로운 출처 섹션 생성
            citation_section = "\n\n**📚 출처:**\n"
            for cit in citations:
                citation_section += f"{cit['number']}. [{cit['title']}]({cit['url']}) - {cit['source']}\n"

            answer_with_links = main_answer + citation_section

        return {
            'answer_with_links': answer_with_links,
            'citations': citations
        }

    def verify_citations(
        self,
        answer: str,
        source_documents: List[Dict]
    ) -> Dict:
        """
        답변의 출처가 실제 문서와 일치하는지 검증

        Returns:
            {
                'verified': True/False,
                'cited_docs': [답변에서 언급한 문서들],
                'available_docs': [실제 제공된 문서들],
                'hallucinated': [실제 없는데 언급된 문서들]
            }
        """

        # 답변에서 언급된 문서들
        cited_docs = self.extract_citations_from_answer(answer)

        # 실제 제공된 문서들
        available_docs = [doc.get('source', '') for doc in source_documents]

        # Hallucination 확인 (없는 출처 언급)
        hallucinated = []
        for cited in cited_docs:
            found = False
            for available in available_docs:
                if cited in available:
                    found = True
                    break
            if not found:
                hallucinated.append(cited)

        return {
            'verified': len(hallucinated) == 0,
            'cited_docs': cited_docs,
            'available_docs': available_docs,
            'hallucinated': hallucinated
        }

3.1 답변 검증 시스템

📖 스토리

수요일 오전, 레이나가 중요한 이슈를 제기했다.

"LLM의 가장 큰 문제는 Hallucination이야," 레이나가 말했다. "문서에 없는 내용을 그럴듯하게 지어내거든."

준혁이 걱정스러운 표정을 지었다. "그럼 잘못된 정보를 줄 수도 있겠네요..."

"맞아. 그래서 검증 시스템이 필요해."

class HallucinationDetector:
    """
    Hallucination 감지 시스템

    답변이 문서에 기반했는지 검증
    """

    def __init__(self, llm_service):
        self.llm = llm_service

    def check_answer_grounding(
        self,
        answer: str,
        source_documents: List[str]
    ) -> Dict:
        """
        답변이 문서에 근거했는지 확인

        방법: 답변의 각 문장이 문서에서 찾아지는지 체크
        """

        # 답변을 문장으로 분리
        import re
        sentences = re.split(r'[.!?]\s+', answer)
        sentences = [s.strip() for s in sentences if len(s.strip()) > 10]

        # 전체 문서 텍스트
        all_docs_text = ' '.join(source_documents)

        results = []
        grounded_count = 0

        for sentence in sentences:
            # 숫자, 출처 표시 등 제외
            if sentence.startswith('출처') or sentence.startswith('**'):
                continue

            # 키워드 추출 (간단히 명사만)
            keywords = [w for w in sentence.split() if len(w) > 2]

            # 문서에서 키워드 매칭 비율
            matched = sum(1 for kw in keywords if kw in all_docs_text)
            match_ratio = matched / len(keywords) if keywords else 0

            is_grounded = match_ratio >= 0.5  # 50% 이상 매칭

            if is_grounded:
                grounded_count += 1

            results.append({
                'sentence': sentence,
                'grounded': is_grounded,
                'match_ratio': match_ratio
            })

        overall_score = grounded_count / len(results) if results else 0

        return {
            'grounding_score': overall_score,
            'total_sentences': len(results),
            'grounded_sentences': grounded_count,
            'details': results,
            'is_reliable': overall_score >= 0.8  # 80% 이상이면 신뢰
        }

    def verify_with_llm(
        self,
        answer: str,
        source_documents: List[str],
        original_query: str
    ) -> Dict:
        """
        LLM을 사용한 검증

        다른 LLM에게 답변이 문서에 기반했는지 물어봄
        """

        verification_prompt = f"""다음 답변이 제공된 문서에만 기반했는지 검증해주세요.

**원본 질문:**
{original_query}

**제공된 문서:**
{chr(10).join(f"[문서 {i+1}] {doc[:200]}..." for i, doc in enumerate(source_documents))}

**생성된 답변:**
{answer}

**검증 기준:**
1. 답변의 모든 사실이 문서에서 확인 가능한가?
2. 문서에 없는 내용을 추측하지 않았는가?
3. 숫자, 날짜, 이름 등이 정확한가?

다음 형식으로 답변해주세요:
검증 결과: [통과/실패]
이유: [구체적 이유]
Hallucination 의심 부분: [있다면 명시]"""

        result = self.llm.generate(verification_prompt, max_tokens=300)

        # 결과 파싱
        verification_text = result['answer']
        is_verified = '통과' in verification_text and '실패' not in verification_text

        return {
            'verified': is_verified,
            'verification_text': verification_text,
            'cost': result['cost']
        }

class DataPulseAnswerGenerator:
    """
    DataPulse 통합 Answer Generation 시스템

    검색 → 프롬프트 → LLM → 검증 → 출처 → 답변
    """

    def __init__(
        self,
        query_understanding_pipeline,
        llm_service,
        prompt_builder: RAGPromptBuilder,
        citation_system: CitationSystem,
        hallucination_detector: HallucinationDetector
    ):
        self.query_pipeline = query_understanding_pipeline
        self.llm = llm_service
        self.prompt_builder = prompt_builder
        self.citation_system = citation_system
        self.detector = hallucination_detector

        # 통계
        self.stats = {
            'total_queries': 0,
            'successful': 0,
            'hallucination_detected': 0,
            'total_cost': 0.0
        }

    def generate_answer(
        self,
        query: str,
        enable_verification: bool = True
    ) -> Dict:
        """
        질문에 대한 답변 생성

        전체 파이프라인:
        1. Query Understanding (Episode 6)
        2. Hybrid Search (Episode 5)
        3. Prompt 생성
        4. LLM 답변 생성
        5. Hallucination 검증
        6. Citation 추가

        Returns:
            {
                'query': 원본 질문,
                'answer': 최종 답변,
                'citations': 출처 목록,
                'verified': 검증 통과 여부,
                'cost': 비용,
                'time': 소요 시간
            }
        """
        import time
        start_time = time.time()

        self.stats['total_queries'] += 1

        print(f"\n{'='*80}")
        print(f"💬 답변 생성 시작")
        print(f"{'='*80}")
        print(f"질문: '{query}'")

        # 1단계: Query Understanding + 검색
        print(f"\n1️⃣ 검색 중...")
        search_result = self.query_pipeline.understand_and_search(
            query,
            top_k=3,
            alpha=0.5
        )

        source_docs = search_result['results']
        print(f"   ✅ {len(source_docs)}개 문서 검색 완료")

        if not source_docs:
            return {
                'query': query,
                'answer': "죄송합니다. 관련 문서를 찾을 수 없습니다.",
                'citations': [],
                'verified': True,
                'cost': 0.0,
                'time': time.time() - start_time
            }

        # 2단계: RAG 프롬프트 생성
        print(f"\n2️⃣ 프롬프트 생성 중...")
        documents_for_prompt = [
            {
                'document': doc['document'],
                'source': f"문서{i+1}",
                'doc_id': f"doc_{doc['doc_idx']}"
            }
            for i, doc in enumerate(source_docs)
        ]

        prompt = self.prompt_builder.build_prompt(
            query,
            documents_for_prompt
        )
        print(f"   ✅ 프롬프트 준비 완료 ({len(prompt)}자)")

        # 3단계: LLM 답변 생성
        print(f"\n3️⃣ LLM 답변 생성 중...")
        llm_result = self.llm.generate(prompt, max_tokens=500, temperature=0.3)

        answer = llm_result['answer']
        cost = llm_result['cost']

        print(f"   ✅ 답변 생성 완료 ({llm_result['usage']['output_tokens']} 토큰)")

        # 4단계: Hallucination 검증
        verified = True
        if enable_verification:
            print(f"\n4️⃣ Hallucination 검증 중...")

            source_texts = [doc['document'] for doc in documents_for_prompt]
            grounding_result = self.detector.check_answer_grounding(
                answer,
                source_texts
            )

            verified = grounding_result['is_reliable']
            grounding_score = grounding_result['grounding_score']

            if verified:
                print(f"   ✅ 검증 통과 (신뢰도: {grounding_score:.0%})")
                self.stats['successful'] += 1
            else:
                print(f"   ⚠️  Hallucination 의심 (신뢰도: {grounding_score:.0%})")
                self.stats['hallucination_detected'] += 1

                # 경고 추가
                answer = f"⚠️ 주의: 이 답변은 신뢰도가 낮을 수 있습니다.\n\n{answer}"

        # 5단계: Citation 추가
        print(f"\n5️⃣ 출처 링크 추가 중...")
        citation_result = self.citation_system.add_citation_links(
            answer,
            documents_for_prompt
        )

        final_answer = citation_result['answer_with_links']
        citations = citation_result['citations']

        print(f"   ✅ {len(citations)}개 출처 링크 추가")

        # 통계 업데이트
        self.stats['total_cost'] += cost

        elapsed_time = time.time() - start_time

        print(f"\n{'='*80}")
        print(f"✅ 답변 생성 완료")
        print(f"{'='*80}")
        print(f"소요 시간: {elapsed_time:.2f}초")
        print(f"비용: ${cost:.6f}")
        print(f"검증: {'✅ 통과' if verified else '⚠️  주의 필요'}")

        return {
            'query': query,
            'answer': final_answer,
            'citations': citations,
            'verified': verified,
            'cost': cost,
            'time': elapsed_time,
            'grounding_score': grounding_result['grounding_score'] if enable_verification else 1.0
        }

    def get_stats(self):
        """통계 조회"""
        return {
            **self.stats,
            'success_rate': (self.stats['successful'] / self.stats['total_queries'] * 100) 
                           if self.stats['total_queries'] > 0 else 0,
            'avg_cost': (self.stats['total_cost'] / self.stats['total_queries'])
                       if self.stats['total_queries'] > 0 else 0
        }

# 통합 시스템 구축
